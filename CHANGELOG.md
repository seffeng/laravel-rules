* 2021-07-01

  * 增加身份证号码验证规则

  ```php
  /**
   * TestRequest.php
   * 表单验证示例
   */
  namespace App\Http\Requests;
  
  use Seffeng\LaravelRules\Rules\IDNumber;
  
class TestRequest
  {
      protected $fillable = ['identify'];
  
      public function rules()
      {
          return [
              'identify' => [
                  'required',
                  new IDNumber()
              ],
          ];
      }
  
      public function attributes()
      {
          return [
              'identify' => '身份证号码',
          ];
      }
  }
  ```
